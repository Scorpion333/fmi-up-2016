	0.Capital or small letter

#include<iostream>
using namespace std;

int main()
{
	char sym;
	cin >> sym;
	
	if (sym >= 'A' && sym <= 'Z')
	{
		cout << "Capital letter." << endl;
	}
	else
	{
		if (sym >= 'a' && sym <= 'z')
			cout << "Small letter." << endl;
		else
			cout << "Not a letter." << endl;
	}

	system("pause");
	return 0;
}

	2.How many digits?

#include<iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int digits = 0;

	for (int i = 1; i <= n; i++)
	{
		char sym;
		cin >> sym;

		if (sym >= '0' && sym <= '9')
		{
			digits++;
		}
	}
	cout << "Digits: " << digits << endl;
	
	system("pause");
	return 0;
}

#include<iostream>
#include<cmath>

int main()
{
	double x1, y1, x2, y2;

	cin >> x1 >> y1;

	8. Biggest and smallest digit

#include<iostream>
using namespace std;

int main()
{
	int n;
	cin >> n;
	
	int max = 0;
	int min = 9;
	int current_digit;
	
	while(n > 0)
	{
		current_digit = n % 10;
		
		if(current_digit > max)
			max = current_digit;
		
		if(current_digit < min)
			min = current_digit;
		
		n = n / 10;
	}
	
	cout << "max: " << max << endl;
	cout << "min: " << min << endl;

	return 0;
}
